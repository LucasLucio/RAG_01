operation calculaCompra
	params
		struct pCompra : out 
		numeric pPagamento : in
	endparams
	
	variables
		struct vProduto
		numeric vSubTotal
		numeric vValorItem
		numeric vAjuste
		numeric vTotal
	endvariables
	
	vSubTotal = 0
	
	forlist vProduto in pCompra->produtos
		;O algoritmo deve calcular o valor total de cada produto multiplicando a quantidade pelo valor unitário
		vValorItem = vProduto->quantidade * vProduto->valorunitarioitem
		;e em seguida somar esses valores para obter o sub-total da compra
		vSubTotal = vSubTotal + vValorItem
	endfor
	
	; 1 = dinheiro desconto de 5
	; 2 = cartao de credito acrescimo de 3,5
	; 3 = cartao de debito acrescimo de 1,5
	
	if (pPagamento = 1)
		vAjuste = vSubTotal * -0.05
	elseif (pPagamento = 2)
		vAjuste = vSubTotal * 0.035
	elseif (pPagamento = 3)
		vAjuste = vSubTotal * 0.015
	endif
	
;	forlist vProduto in pCompra->produtos
;		;O algoritmo deve calcular o valor total de cada produto multiplicando a quantidade pelo valor unitário
;		vValorItem = vProduto->quantidade * vProduto->valorunitarioitem
;		;e em seguida somar esses valores para obter o sub-total da compra
;		vSubTotal = vSubTotal + vValorItem
;	endfor
		
	vTotal = vSubTotal + vAjuste
	
	pCompra->total = vTotal
end
