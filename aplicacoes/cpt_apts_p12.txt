operation validarNota
params
	numeric nota1 : IN
	numeric nota2 : IN
	numeric nota3 : IN
	numeric nota4 : IN
	numeric nma   : IN
	struct  estado : OUT
endparams
variables
	numeric mediaTotal
	
endvariables
	
	estado = $newstruct

	if(nota1 < 0 | nota1 > 10 | nota2 < 0 | nota2 > 10 | nota3 < 0 | nota3 > 10 |nota4 < 0 | nota4 > 10 |nma   < 0 | nma   > 10 )
		estado->NOTA = -1
		estado->APROVADO = 0
		return(-1)
	endif
	
	mediaTotal = (nota1 + nota2 + nota3 + nota4) / 4
	
	if(mediaTotal >= nma)
		estado->NOTA = mediaTotal
		estado->APROVADO = 1
	else
		estado->NOTA = mediaTotal
		estado->APROVADO = 0
	endif
	
	return(0)
end

operation compra
	params
		struct ListaCompra  : IN
		numeric tpPagamento : IN
		struct TotalPagar   : OUT
	endparams
	variables
		numeric qtCompras, vindex, vlUnitario, qtProduto, vltotal
		string nmProduto
		struct aux
	endvariables
	
	qtCompras = ListaCompra->OCC->$collsize	
	
	TotalPagar = $newstruct
	
	vltotal = 0
	for vindex = 1 to qtCompras
		aux = ListaCompra->OCC{vindex}
		vlUnitario = aux->UNITARIO
		qtProduto = aux->QUANTIDADE
		nmPRODUTO = aux->NOME
		TotalPagar->OCC{vindex} = $newstruct
		TotalPagar->OCC{vindex}->NOME = nmPRODUTO
		TotalPagar->OCC{vindex}->QUANTIDADE = qtProduto
		TotalPagar->OCC{vindex}->TOTAL = (qtProduto * vlUnitario)
		vltotal += (qtProduto * vlUnitario)
	endfor
	
	TotalPagar->SUBTOTAL = vltotal
	
	selectcase tpPagamento
	case 1
		;desconto 5%
		TotalPagar->DESCONTO = "5%"
		TotalPagar->TOTAL = vltotal - (vltotal*0.05)
		
	case 2
		;acrescimo 3.5%
		TotalPagar->TOTAL = vltotal + (vltotal*0.035)
		TotalPagar->ACRESCIMO = "3,5%"
	case 3
		;acrescimo 1.5%
		TotalPagar->TOTAL = vltotal + (vltotal*0.015)
		TotalPagar->ACRESCIMO = "1,5%"
	endselectcase
	
	
	return(0)
end
